        function initQuiz() {

            const container = document.getElementById('quiz-container');

            quizData.forEach((item, index) => {

                    </p>

        window.toggleHint = function(index) {

            const hint = document.getElementById(`hint-${index}`);

            hint.classList.toggle('hidden-content');

        }


        window.checkAnswer = function(qIndex, oIndex) {

            if (answeredQuestions.has(qIndex)) return;

            

            const correct = quizData[qIndex].answer === oIndex;

            const feedback = document.getElementById(`feedback-${qIndex}`);

            const feedbackText = document.getElementById(`feedback-text-${qIndex}`);

            const options = document.querySelectorAll(`[id^="btn-${qIndex}-"]`);


            // Disable all options for this question

            options.forEach((btn, i) => {

                btn.disabled = true;

                if (i === quizData[qIndex].answer) {

                    btn.classList.add('border-emerald-500', 'bg-emerald-50', 'text-emerald-700');

                } else if (i === oIndex && !correct) {

                    btn.classList.add('border-red-500', 'bg-red-50', 'text-red-700');

                }

                btn.classList.remove('hover:border-indigo-300', 'hover:bg-indigo-50');

            });


            if (correct) {

                currentScore++;

                feedback.classList.add('bg-emerald-100', 'border', 'border-emerald-200');

                feedbackText.innerText = "✓ Correct!";

                feedbackText.classList.add('text-emerald-700');

            } else {

                feedback.classList.add('bg-red-100', 'border', 'border-red-200');

                feedbackText.innerText = "✗ Incorrect";

                feedbackText.classList.add('text-red-700');

            }


            feedback.classList.remove('hidden-content');

            answeredQuestions.add(qIndex);

            updateScore();


            if (answeredQuestions.size === quizData.length) {

                document.getElementById('completion-card').classList.remove('hidden-content');

                document.getElementById('completion-card').scrollIntoView({ behavior: 'smooth' });

            }

        }


        function updateScore() {

            const scoreDisp = document.getElementById('score-display');

            const progress = document.getElementById('progress-bar');

            const percent = (answeredQuestions.size / quizData.length) * 100;

            

            scoreDisp.innerText = `${currentScore}/10`;

            progress.style.width = `${percent}%`;

        }


        initQuiz();

    </script>
